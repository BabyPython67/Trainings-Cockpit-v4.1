// Hex → HSL → Hex, um im Dark Mode JEDE hartcodierte Textfarbe automatisch hell genug zu machen —
// unabhängig davon, welcher Ton irgendwo im Code verwendet wird. Verhindert diese Bugklasse dauerhaft.
function hexToHsl(hex) {
  const r = parseInt(hex.slice(1, 3), 16) / 255, g = parseInt(hex.slice(3, 5), 16) / 255, b = parseInt(hex.slice(5, 7), 16) / 255;
  const max = Math.max(r, g, b), min = Math.min(r, g, b); let h = 0, s = 0; const l = (max + min) / 2;
  if (max !== min) {
    const d = max - min; s = l > 0.5 ? d / (2 - max - min) : d / (max + min);
    if (max === r) h = (g - b) / d + (g < b ? 6 : 0); else if (max === g) h = (b - r) / d + 2; else h = (r - g) / d + 4;
    h /= 6;
  }
  return [h * 360, s * 100, l * 100];
}
function hslToHex(h, s, l) {
  s /= 100; l /= 100;
  const k = (n) => (n + h / 30) % 12;
  const a = s * Math.min(l, 1 - l);
  const f = (n) => l - a * Math.max(-1, Math.min(k(n) - 3, Math.min(9 - k(n), 1)));
  const toHex = (x) => Math.round(255 * x).toString(16).padStart(2, "0");
  return `#${toHex(f(0))}${toHex(f(8))}${toHex(f(4))}`;
}
function safeText(dark, hex) {
  if (!dark || !hex || hex[0] !== "#") return hex;
  try { const [h, s, l] = hexToHsl(hex); return hslToHex(h, Math.min(Math.max(s, 40), 85), Math.max(l, 72)); } catch (_) { return hex; }
}

  return (props) => {
    const { viewBox } = props;
    if (!viewBox) return null;
    const fontSize = 9, padX = 5, h = 13;
    const w = Math.round(text.length * 5.4 + padX * 2);
    const x = bias === "vtop" ? Math.max(2, viewBox.x - w / 2) : viewBox.x + viewBox.width - w - 3;
    const y = bias === "above" ? viewBox.y - h - 3 : bias === "below" ? viewBox.y + 3 : bias === "vtop" ? viewBox.y - h - 2 : viewBox.y - h / 2;

      <g>

      </g>
    );
  };

function weekMeta(week) {

/* ===== v5.0: Krankheits-Logik — Phasen erkennen, Progression verzögern, Planende steuern ===== */
const ILLNESS_RAMP_DAYS = 3; // Wiedereinstiegs-Fenster nach Krankheitsende: Umfang vor Intensität
function illnessPhaseForDate(date, illnessLog) {
  const t = atMidnight(date).getTime();
  for (const e of illnessLog || []) {
    if (!e || !e.start) continue;
    const s = new Date(e.start + "T00:00:00").getTime();
    const en = e.end ? new Date(e.end + "T00:00:00").getTime() : atMidnight(new Date()).getTime(); // offenes Ende = noch krank
    if (t >= s && t <= en) return "sick";
    if (e.end && t > en && t <= en + ILLNESS_RAMP_DAYS * DAY_MS) return "ramp";
  }
  return null;
}
// Kalenderwochen mit >=3 Krank-/Aufbau-Tagen zählen nicht als normale Trainingswoche
function lostTrainingWeeks(illnessLog) {
  if (!illnessLog || !illnessLog.length) return 0;
  let lost = 0;
  for (let w = 1; w <= TOTAL_WEEKS; w++) {
    const monday = mondayOfWeek(w);
    let affected = 0;
    for (let i = 0; i < 7; i++) if (illnessPhaseForDate(new Date(monday.getTime() + i * DAY_MS), illnessLog)) affected++;
    if (affected >= 3) lost++;
  }
  return lost;
}
function lastAffectedWeek(illnessLog) {
  if (!illnessLog || !illnessLog.length) return 0;
  let last = 0;
  for (let w = 1; w <= TOTAL_WEEKS; w++) {
    const monday = mondayOfWeek(w);
    for (let i = 0; i < 7; i++) if (illnessPhaseForDate(new Date(monday.getTime() + i * DAY_MS), illnessLog)) { last = w; break; }
  }
  return last;
}
// Nach der Krankheit hinkt der Block-Inhalt dem Kalender um die verlorenen Wochen hinterher —
// Kalenderwochen-Nummern und der Turniertermin bleiben unangetastet.
function effectiveWeek(week, illnessLog) {
  const lost = lostTrainingWeeks(illnessLog);
  if (!lost || week <= lastAffectedWeek(illnessLog)) return week;
  return Math.max(1, week - lost);
}
// "fixed": Testtermin bleibt, der Endblock wird komprimiert · "dynamic": Plan verlängert sich um die verlorenen Wochen
function planTotalWeeks(illnessLog, planEndMode) {
  return TOTAL_WEEKS + (planEndMode === "dynamic" ? lostTrainingWeeks(illnessLog) : 0);
}
function testDateFor(illnessLog, planEndMode) {
  return new Date(mondayOfWeek(planTotalWeeks(illnessLog, planEndMode)).getTime() + 2 * DAY_MS);
}

    // v5.0: Krank-/Aufbau-Markierung — Einheiten bleiben sichtbar & abhakbar (uid unverändert), nur Empfehlung/Badge ändern sich
    if (sickPhase === "sick") items.forEach((it) => { if (it.load > 0) { it.illness = "sick"; it.desc += " · Krank gemeldet — Pause empfohlen, höchstens sehr locker"; } });
    else if (sickPhase === "ramp") items.forEach((it) => {

      else if (it.sport === "strength") { it.illness = "ramp"; it.desc += " · Aufbau nach Krankheit: nur Mobility, nichts Schweres"; }
    });
    if (items.length === 0) items.push({ uid: mkuid(day, "rest"), sport: "rest", label: "Ruhetag", dur: "Erholung", desc: day === "So" ? "Aktive Erholung optional: Spaziergang, Dehnen" : "Kein Training geplant", load: 0 });
    return { day, items, load: items.reduce((s, i) => s + (i.load || 0), 0) };
  });
}

// Einmalige Migration: alte positionsbasierte Häkchen-Keys ("3-Mi-1") auf stabile uids ("w3-Mi-vb") umziehen
function migrateDone(done) {
  const out = {}; let changed = false; const cache = {};
  Object.entries(done || {}).forEach(([k, v]) => {
    const m = k.match(/^(\d+)-(Mo|Di|Mi|Do|Fr|Sa|So)-(\d+)$/);
    if (!m) { out[k] = v; return; }
    changed = true;
    if (!v) return; // nicht gesetzte alte Keys einfach verwerfen
    const week = Number(m[1]);
    const plan = cache[week] || (cache[week] = buildWeekPlan(week)); // Standard-Slots: entspricht dem Stand, mit dem die alten Keys entstanden
    const item = plan.find((d) => d.day === m[2])?.items[Number(m[3])];
    if (item?.uid) out[item.uid] = true;
  });
  return { done: out, changed };
}

function usePersistentData() {
  const [data, setData] = useState(DEFAULT_DATA);
  const [loaded, setLoaded] = useState(false);
  const [savedAt, setSavedAt] = useState(null);
  useEffect(() => {
    let alive = true;
    (async () => {
      if (hasStorage) { try {
        const res = await window.storage.get(STORAGE_KEY, false);
        if (alive && res?.value) {
          let next = { ...DEFAULT_DATA, ...JSON.parse(res.value) };
          const mig = migrateDone(next.done);
          if (mig.changed) next = { ...next, done: mig.done };
          setData(next); setSavedAt(new Date());
        }
      } catch (_) {} }
      if (alive) setLoaded(true);
    })();
    return () => { alive = false; };
  }, []);
  useEffect(() => {
    if (!loaded || !hasStorage) return;
    const t = setTimeout(() => { window.storage.set(STORAGE_KEY, JSON.stringify(data), false).then(() => setSavedAt(new Date())).catch(() => {}); }, 250);
    return () => clearTimeout(t);
  }, [data, loaded]);
  return [data, setData, loaded, savedAt];
}

// Nap "zählt", wenn effektive Minuten (Dauer × Gefühls-Faktor) mind. 15 erreichen —
// ein groggy-Nap braucht also mehr Rohminuten, um überhaupt als erholsam zu gelten.
function napEffectiveMin(nap) {
  if (!nap || !nap.min) return 0;
  const factor = nap.feeling === "groggy" ? 0.5 : nap.feeling === "erfrischt" ? 1 : 0.85;
  return nap.min * factor;
}
function readinessStatus(weekLoad, avgSleep, readiness, cutFirst, nap) {
  const cut = cutFirst || "optionale Einheiten";
  let level;
  if (!avgSleep && !readiness) level = "none";
  else if ((avgSleep && avgSleep < 6) || readiness <= 2 || (avgSleep && avgSleep < 7 && weekLoad > 40)) level = "red";
  else if (avgSleep && avgSleep >= 7.5 && readiness >= 4) level = "green";
  else level = "yellow";

  // Option C: Nap kann die Ampel um genau eine Stufe anheben (Rot→Gelb oder Gelb→Grün) —
  // nie zwei Stufen auf einmal, nie aus "Noch keine Daten" heraus. Schlechter Nachtschlaf
  // bleibt dadurch sichtbar, statt von der Ampel wegrationalisiert zu werden.
  let napBoost = false;
  if (level !== "none" && napEffectiveMin(nap) >= 15) {
    const idx = LEVEL_ORDER.indexOf(level);
    if (idx >= 0 && idx < LEVEL_ORDER.length - 1) { level = LEVEL_ORDER[idx + 1]; napBoost = true; }
  }

  const meta = LEVEL_META[level];
  let hint;
  if (level === "none") hint = "Trag einen Check-in ein, um deine Bereitschaft zu sehen.";
  else if (level === "red") hint = `Schlaf/Erholung niedrig. Streiche zuerst ${cut} und halte die harten Tage locker.`;
  else if (level === "green") hint = napBoost ? "Der Nap hat dich hochgeholt — die Grundlage bleibt dünn, priorisiere trotzdem den Nachtschlaf." : "Gut erholt. Plan wie vorgesehen umsetzen.";

  return { level, napBoost, ...meta, hint };
}

// Riegel-Formel: T2 = T1 · (D2/D1)^1.06 — Standard-Hochrechnung für Laufzeiten
function riegel5k(dist, sec) {
  if (!dist || !sec || dist < 0.75) return null;
  const t = Math.round(sec * Math.pow(5 / dist, 1.06));
  return t >= 720 && t <= 3900 ? t : null; // Plausibilität: 12–65 Min
}

// Arbeitsrunden (schnelles Cluster) von Trab-/Standpausen trennen — Basis für Intervalle & Tempo
function classifyLaps(laps) {
  const usable = (laps || []).filter((l) => l.pace > 0 && ((l.dist || 0) >= 0.15 || (l.sec || 0) >= 45));
  if (usable.length < 2) return null;
  const fastest = Math.min(...usable.map((l) => l.pace));
  const work = usable.filter((l) => l.pace <= fastest * 1.12);
  if (!work.length || work.length === usable.length) return null; // kein klarer Arbeit/Pause-Kontrast (z. B. gleichmäßiger Lauf)
  const dist = work.reduce((s, l) => s + (l.dist || 0), 0);
  const sec = work.reduce((s, l) => s + (l.sec || (l.dist ? l.pace * l.dist : 0)), 0);
  if (dist < 0.4 || !sec) return null;
  const hrs = work.filter((l) => l.hr).map((l) => l.hr);
  return { work, dist: Math.round(dist * 100) / 100, sec: Math.round(sec), pace: sec / dist, hr: hrs.length ? Math.round(hrs.reduce((a, b) => a + b, 0) / hrs.length) : null, reps: work.length, fastest, slowest: Math.max(...work.map((l) => l.pace)) };
}
// Fasst einen Satz Runden zu einem Abschnitt zusammen (Distanz/Zeit/Pace/HF).
// extractLaps() liefert nur "Laufen"-Runden — Warm up/Cool Down sind hier also schon draußen.
function aggregateLaps(laps) {
  const usable = (laps || []).filter((l) => l.pace > 0 && ((l.dist || 0) >= 0.15 || (l.sec || 0) >= 45));
  if (!usable.length) return null;
  const dist = usable.reduce((s, l) => s + (l.dist || 0), 0);
  const sec = usable.reduce((s, l) => s + (l.sec || (l.dist ? l.pace * l.dist : 0)), 0);
  if (dist < 0.4 || !sec) return null;
  const hrs = usable.filter((l) => l.hr).map((l) => l.hr);
  return { dist: Math.round(dist * 100) / 100, sec: Math.round(sec), pace: sec / dist, hr: hrs.length ? Math.round(hrs.reduce((a, b) => a + b, 0) / hrs.length) : null, reps: usable.length, work: new Set(usable.map((l) => l.lap)) };
}
// ZENTRAL: der auswertungs-relevante Abschnitt eines Laufs — Basis für Prognose, Pace-Chart,
// Bestzeiten, Pace-HF/VO2max und Splits. Nutzt IMMER die Runden (nicht die verwässerte
// Gesamt-Statistik inkl. Warm up/Cool Down), sofern welche da sind.
// - Intervalle: Arbeitsrunden vom Trab trennen (Kontrast); ohne Kontrast = alle Runden Arbeit.
// - Tempo/Test/Easy/Lang: der Lauf-Abschnitt (Warm up/Cool Down sind via extractLaps schon raus).
// - Ohne Runden: Rückfall auf die Gesamt-Statistik.
function runEffort(run) {
  const laps = run.laps, hasLaps = laps && laps.length >= 2;
  if (run.type === "int") {
    if (hasLaps) {
      const c = classifyLaps(laps);
      if (c) return { dist: c.dist, sec: c.sec, pace: c.pace, hr: c.hr, reps: c.reps, work: new Set(c.work.map((l) => l.lap)), basis: `${c.reps}× Arbeitsrunden` };
      const a = aggregateLaps(laps);
      if (a) return { ...a, basis: `${a.reps}× Intervalle` };
    }
    return run.imported ? null : { dist: run.dist, sec: run.sec, pace: run.pace, hr: run.hrAvg ?? null, basis: "Intervall (manuell)" };
  }
  const label = run.type === "tempo" ? "Tempo-Block" : run.type === "test" ? "5K-Test" : "Lauf-Abschnitt";
  if (hasLaps) { const a = aggregateLaps(laps); if (a) return { ...a, basis: label }; }
  return { dist: run.dist, sec: run.sec, pace: run.pace, hr: run.hrAvg ?? null, basis: run.type === "tempo" ? "Tempo" : run.type === "test" ? "5K-Test" : "Gesamt" };
}
// Renn-repräsentativer Qualitäts-Effort je Lauftyp — einzige Basis der 5K-Prognose.
// Easy/Lange fließen bewusst NICHT ein: Riegel setzt Wettkampfanstrengung voraus.
function qualityEffort(run) {
  if (run.type !== "test" && run.type !== "int" && run.type !== "tempo") return null;
  const e = runEffort(run);
  return e ? { dist: e.dist, sec: e.sec, basis: e.basis } : null;
}
// Effektive Pace für Charts/Bestzeiten — immer der relevante Abschnitt (runEffort), sonst Gesamt.
function effectivePace(run) {
  const e = runEffort(run);
  return e ? e.pace : run.pace;
}

const hrMaxTanaka = (age) => Math.round(208 - 0.7 * (age || 18)); // konsistent mit den HF-Zonen des Laufplans (≈196)

// ACSM-Laufformel (VO2 bei Pace) + Swain-Umrechnung (%HFmax → %VO2max) → Schätzwert, Tendenz zählt
function estimateVo2max(paceSecPerKm, hrAvg, hrMax) {
  if (!paceSecPerKm || !hrAvg || !hrMax) return null;
  const pctHr = hrAvg / hrMax;
  if (pctHr < 0.6 || pctHr > 0.96) return null; // zu locker/zu hart → Umrechnung unzuverlässig
  const vo2 = 3.5 + 0.2 * (60000 / paceSecPerKm);
  const pctVo2 = (pctHr * 100 - 37) / 0.64 / 100;
  if (pctVo2 <= 0.3) return null;
  const est = vo2 / pctVo2;
  return est >= 25 && est <= 80 ? Math.round(est * 10) / 10 : null;
}

// v5.0: Krank-Tage zählen nicht mit — eine Krankheitswoche bricht die Serie nicht.
function weekFulfilled(w, done, vbSlots, illnessLog, planEndMode) {
  const items = buildWeekPlan(w, undefined, vbSlots, illnessLog, planEndMode).flatMap((d) => d.items).filter((i) => i.load > 0 && i.illness !== "sick");
  if (!items.length) return true; // komplett entschuldigte Woche: Serie läuft weiter
  return items.filter((i) => done[i.uid]).length / items.length >= 0.8;
}
function computeStreak(currentWeek, done, vbSlots, illnessLog, planEndMode) {
  let s = 0;
  for (let w = currentWeek - 1; w >= 1; w--) { if (weekFulfilled(w, done, vbSlots, illnessLog, planEndMode)) s++; else break; }
  if (weekFulfilled(currentWeek, done, vbSlots, illnessLog, planEndMode)) s++;
  return s;
}

  if (test) return { tone: "info", text: "Testtag! Nur 10 Min locker einlaufen, 2–3 Steigerungen — dann Bestzeit jagen." };
  if (!works.length) return { tone: "rest", text: "Heute ist Ruhetag — der Fortschritt passiert in der Erholung. Schlaf hat Prio." };
  const hardRun = works.find((i) => i.sport === "run" && (i.type === "int" || i.type === "tempo"));
  const lower = works.find((i) => i.sport === "strength" && i.key === "lower");
  const vb = works.find((i) => i.sport === "vb");
  const soreLegs = (latest?.soreness || 0) >= 4;
  if (level === "red") {
    if (hardRun) return { tone: "warn", text: `${hardRun.label} heute ersetzen: 20–25 Min Easy (HF 130–140). Der harte Reiz bringt ohne Erholung nichts.` };
    if (lower) return { tone: "warn", text: "Schwere Beine heute streichen oder auf 20 Min Mobility reduzieren." };
    if (vb) return { tone: "warn", text: "Volleyball ok — aber Sprungvolumen bewusst klein halten und danach früh ins Bett." };
    return { tone: "warn", text: "Nur lockere Bewegung heute. Schlaf nachholen ist die beste Trainingseinheit." };
  }
  if (soreLegs && (hardRun || lower || vb)) return { tone: "warn", text: "Beine melden starken Muskelkater: Sprünge & Intervalle heute reduzieren, Technik/Oberkörper geht klar." };
  if (level === "yellow") {
    if (hardRun) return { tone: "mid", text: `${hardRun.label} heute leicht kürzen (z. B. eine Wiederholung weniger) und auf die HF-Obergrenze achten.` };
    return { tone: "mid", text: "Plan machbar — locker angehen und nach der Einheit ehrlich in den Check-in." };
  }
  if (level === "green") {
    const main = hardRun || lower || vb || works[0];
    return { tone: "go", text: `Grünes Licht — volle Qualität bei: ${main.label}${main.time ? ` (${main.time})` : ""}.` };
  }
  return { tone: "none", text: "Trag im Recovery-Tab einen Check-in ein — dann bekommst du hier jeden Tag eine konkrete Empfehlung." };
}
/* ===================== 2.0: Undo beim Löschen ===================== */
function useUndoDelete(setData) {
  const [undo, setUndo] = useState(null);
  const timer = useRef(null);
  const trigger = (label, listKey, item) => {
    setUndo({ label, listKey, item });
    clearTimeout(timer.current);
    timer.current = setTimeout(() => setUndo(null), 6000);
  };
  const doUndo = () => {
    setUndo((u) => {
      if (u) setData((d) => ({ ...d, [u.listKey]: [...d[u.listKey], u.item] }));
      return null;
    });
    clearTimeout(timer.current);
  };
  return { undo, trigger, doUndo };
}
function UndoToast({ undo, onUndo }) {
  if (!undo) return null;
  return (
    <div className="fixed left-1/2 -translate-x-1/2 z-30 rounded-xl bg-slate-900 text-white text-xs font-semibold px-4 py-3 flex items-center gap-4 shadow-lg" style={{ bottom: "calc(1rem + env(safe-area-inset-bottom, 0px))" }}>
      <span>{undo.label} gelöscht</span>
      <button onClick={onUndo} className="text-teal-300 font-bold">Rückgängig</button>
    </div>
  );
}

function guessRunType(paceSec, dist, sec) {
  if (paceSec == null) return "easy";
  if (dist >= 4.85 && dist <= 5.25 && paceSec <= 305) return "test";
  if (paceSec <= 292) return "int";
  if (paceSec <= 322) return "tempo";
  if (dist >= 7.5 || sec >= 45 * 60) return "long";
  return "easy";
}
// v3.3: Lauftyp beim Import aus dem Trainingsplan ableiten statt nur per Pace zu raten.
// buildWeekPlan ist die einzige Quelle der Wahrheit — damit greifen automatisch auch
// Taper (Turnierwoche: Di-Intervalle -> Shakeout/easy), Regenerationswochen und die Testwoche.
const __planRunCache = {};
function plannedRunFor(dateISO, illnessLog, planEndMode) {
  if (!dateISO) return null;
  const d = new Date(dateISO + "T12:00:00"); // Mittag: keine Zeitzonen-Kippkante
  if (isNaN(d) || atMidnight(d) < PLAN_START) return null;
  const week = Math.round((mondayOf(d) - PLAN_START) / (7 * DAY_MS)) + 1; // bewusst NICHT geklemmt: außerhalb des Plans -> null
  if (week < 1 || week > planTotalWeeks(illnessLog, planEndMode)) return null;
  const day = DAYS[(d.getDay() + 6) % 7];
  const key = `${week}|${lostTrainingWeeks(illnessLog)}|${planEndMode === "dynamic" ? "d" : "f"}`; // Cache invalidiert sich, wenn sich der Krankheits-Verzug ändert
  const plan = __planRunCache[key] || (__planRunCache[key] = buildWeekPlan(week, undefined, undefined, illnessLog, planEndMode));
  const run = plan.find((x) => x.day === day)?.items.find((i) => i.sport === "run");
  return run ? { week, day, type: run.type } : null;
}
const isQualityType = (t) => t === "int" || t === "tempo" || t === "test";
// Typ-Zuordnung beim Import: Plan gewinnt, Pace ist nur Fallback.
// mismatch warnt nur bei grobem Widerspruch (Plan locker, gelaufen schnell — oder umgekehrt);
// Feinunterschiede wie Tempo vs. Intervalle entscheidet allein der Plan.
function assignRunType(dateISO, paceSec, dist, sec, illnessLog, planEndMode) {
  const planInfo = plannedRunFor(dateISO, illnessLog, planEndMode);
  const paceGuess = guessRunType(paceSec, dist, sec);
  if (!planInfo) return { type: paceGuess, source: "pace", paceGuess, planInfo: null, mismatch: false };
  const mismatch = isQualityType(planInfo.type) !== isQualityType(paceGuess);
  return { type: planInfo.type, source: "plan", paceGuess, planInfo, mismatch };
}
function pickCol(row, keys) {
  const lower = {};
  Object.keys(row).forEach((k) => { lower[k.toLowerCase().trim()] = row[k]; });
  for (const key of keys) {
    const hit = Object.keys(lower).find((k) => k.includes(key));
    if (hit && lower[hit] !== "" && lower[hit] != null) return lower[hit];
  }
  return null;
}
function parseGarminDate(v) {
  if (!v) return null;
  const s = String(v).trim();
  const iso = s.match(/^(\d{4})-(\d{2})-(\d{2})/);
  if (iso) return `${iso[1]}-${iso[2]}-${iso[3]}`;
  const de = s.match(/^(\d{1,2})\.(\d{1,2})\.(\d{4})/);
  if (de) return `${de[3]}-${String(de[2]).padStart(2, "0")}-${String(de[1]).padStart(2, "0")}`;
  return null;
}
// Zahlen robust parsen: erkennt "5,00" (DE-Komma), "4.07" (Punkt als Dezimaltrenner) und "1.234,56" (Punkt=Tausender)
function parseNumberLoose(v) {
  if (v == null) return NaN;
  let s = String(v).trim();
  if (!s) return NaN;
  if (s.includes(",") && s.includes(".")) s = s.replace(/\./g, "").replace(",", ".");
  else if (s.includes(",")) s = s.replace(",", ".");
  return Number(s);
}
// Erkennt die Zusammenfassungs-/Übersichtszeile eines Garmin-Einzelaktivitäts- bzw. Rundenexports
// (Format: Intervall/Abschnitttyp/Runde/Zeit/Distanz je Runde, ohne Datumsspalte)
function findSummaryRow(rows) {
  return rows.find((row) => Object.values(row).some((v) => /^(übersicht|summary|total|gesamt)$/i.test(String(v).trim()))) || null;
}
function numOrNull(v) { const n = parseNumberLoose(v); return v == null || v === "" || isNaN(n) ? null : n; }
// Zusatzmetriken, die Garmin-Exporte (Listen- wie Rundenexport) oft mitliefern — bislang ungenutzt, jetzt für smarte Charts nutzbar
function extractExtras(row) {
  const hrAvg = numOrNull(pickCol(row, ["ø herzfrequenz", "avg heart", "average heart", "durchschnittliche hf"]));
  const hrMax = numOrNull(pickCol(row, ["maximale herzfrequenz", "max heart", "max hf"]));
  const elevGain = numOrNull(pickCol(row, ["anstieg gesamt", "elevation gain", "positiver höhenunterschied"]));
  const cadence = numOrNull(pickCol(row, ["ø schrittfrequenz", "avg cadence", "average cadence"]));
  const strideLen = numOrNull(pickCol(row, ["ø schrittlänge", "avg stride", "average stride"]));
  const calories = numOrNull(pickCol(row, ["kalorien", "calories"]));
  const out = {};
  if (hrAvg != null) out.hrAvg = Math.round(hrAvg);
  if (hrMax != null) out.hrMax = Math.round(hrMax);
  if (elevGain != null) out.elevGain = Math.round(elevGain);
  if (cadence != null) out.cadence = Math.round(cadence);
  if (strideLen != null) out.strideLen = Math.round(strideLen * 100) / 100;
  if (calories != null) out.calories = Math.round(calories);
  return out;
}
// Einzelne Runden/Splits aus dem Garmin-Rundenexport (nur echte Laufabschnitte, keine Warm-up/Cool-Down/Sammel-Zeilen)
function extractLaps(rows) {
  const laps = [];
  rows.forEach((row) => {
    const abschnitt = pickCol(row, ["abschnitttyp", "lap type"]);
    const runde = pickCol(row, ["runde", "lap"]);
    if (!abschnitt || !/^(laufen|run)$/i.test(String(abschnitt).trim())) return;
    if (!runde || !/^\d+$/.test(String(runde).trim())) return; // Bereiche wie "2 - 3" überspringen, nur Einzelrunden
    const dist = numOrNull(pickCol(row, ["distanz", "distance"]));
    const sec = parseTimeToSec(pickCol(row, ["zeit", "time"]));
    const hr = numOrNull(pickCol(row, ["ø herzfrequenz", "avg heart"]));
    if (!dist || !sec) return;
    laps.push({ lap: Number(runde), dist: Math.round(dist * 100) / 100, sec, pace: sec / dist, hr: hr != null ? Math.round(hr) : null });
  });
  return laps.sort((a, b) => a.lap - b.lap);
}

// Splits v4: Runden-Liste statt Balkendiagramm. Jede Runde eine Zeile mit exakter Pace,
// Abweichungsbalken relativ zum Ø Arbeit (links = schneller, rechts = langsamer) und HF.
// Kein Achsen-/Kontrast-/Clipping-Risiko, exakte Werte direkt lesbar, 380px-tauglich.
function SplitsChart({ laps, run, illnessLog }) {
  const dark = useDark();
  const eff = run.type === "int" || run.type === "tempo" ? runEffort(run) : null;
  const workSet = eff && eff.work ? eff.work : null;
  const refPace = eff ? eff.pace : run.pace;
  const w = planWeekOf(new Date(run.date + "T12:00:00"), illnessLog);
  const target = (run.type === "int" || run.type === "tempo") && w >= 1 && w <= TOTAL_WEEKS ? Math.round(targetPaceSec(run.type, effectiveWeek(w, illnessLog))) : null;
  const rows = laps.map((l) => {
    const isWork = workSet ? workSet.has(l.lap) : true; // ohne eff (Easy/Lang): alles regulär
    return { lap: l.lap, pace: l.pace, hr: l.hr, dist: l.dist, isWork, dev: isWork ? l.pace - refPace : null };
  });
  // Skala: größte Abweichung der Arbeitsrunden, mind. ±10 s — kleine Streuung wird NICHT künstlich
  // aufgeblasen: ein konstanter Lauf soll auch konstant aussehen.
  const maxDev = Math.max(10, ...rows.filter((r) => r.dev != null).map((r) => Math.abs(r.dev)));
  const label = eff ? (run.type === "int" ? `${eff.reps}× Arbeit` : "Tempo-Block") + ` Ø ${secToMMSS(Math.round(eff.pace))}/km` : `Ø ${secToMMSS(Math.round(refPace))}/km`;
  const muted = dark ? "#64748b" : "#94a3b8", strong = dark ? "#e2e8f0" : "#0f172a";
  return (
    <div className="px-4 pb-3 -mt-1">
      <div className="flex items-center gap-2 mb-1.5 text-[10px]">
        <span className="font-semibold" style={{ color: safeText(dark, "#0d9488") }}>{label}</span>
        {target && <span className="rounded-full px-1.5 py-px font-semibold" style={{ background: softBg(dark, "#8b5cf6", "#f3e8ff"), color: safeText(dark, "#7c3aed") }}>Ziel {secToMMSS(target)}</span>}
        <span style={{ color: muted }}>◄ schneller · langsamer ►</span>
      </div>
      <div className="space-y-1">
        {rows.map((r) => {
          const devW = r.dev != null ? Math.min(Math.abs(r.dev) / maxDev, 1) * 50 : 0;
          const faster = r.dev != null && r.dev < 0;
          return (
            <div key={r.lap} className="flex items-center gap-2" style={{ opacity: r.isWork ? 1 : 0.55 }}>
              <span className="w-6 shrink-0 text-[10px] font-num" style={{ color: muted }}>R{r.lap}</span>
              <span className="w-11 shrink-0 text-xs font-num font-semibold" style={{ color: r.isWork ? strong : muted }}>{secToMMSS(Math.round(r.pace))}</span>
              <div className="relative flex-1 h-3.5 rounded" style={{ background: dark ? "#1c2940" : "#f1f5f9" }}>
                <div className="absolute inset-y-0 left-1/2 w-px" style={{ background: dark ? "#3b4a63" : "#cbd5e1" }} />
                {r.dev != null ? (
                  <div className="absolute inset-y-0.5 rounded-sm" style={faster
                    ? { right: "50%", width: `${devW}%`, background: "#0d9488" }
                    : { left: "50%", width: `${devW}%`, background: r.dev > 15 ? "#f59e0b" : "#818cf8" }} />
                ) : (
                  <span className="absolute inset-0 flex items-center justify-center text-[9px]" style={{ color: muted }}>Pause · {r.dist} km</span>
                )}
              </div>
              <span className="w-9 shrink-0 text-right text-[10px] font-num" style={{ color: r.hr ? safeText(dark, "#e11d48") : muted }}>{r.hr ? `${r.hr} ♥` : "–"}</span>
            </div>
          );
        })}
      </div>
    </div>
  );
}

function RunProgress({ data, setData, currentWeek }) {
  const dark = useDark();
  const [type, setType] = useState("int");
  const [form, setForm] = useState(() => ({ date: todayISO(), type: plannedRunFor(todayISO(), data.illnessLog, data.planEndMode)?.type ?? "int", dist: "", time: "", typeTouched: false })); // v3.3: Typ aus Plan vorbelegt
  const [importMsg, setImportMsg] = useState(null);
  const [pending, setPending] = useState(null); // Vorschau eines erkannten Einzellaufs, wartet auf Datums-Bestätigung
  const [openRun, setOpenRun] = useState(null); // welcher Lauf hat seine Splits aufgeklappt
  const fileRef = useRef(null);

  const add = () => { const sec = parseTimeToSec(form.time), dist = Number(String(form.dist).replace(",", ".")); if (!sec || !dist) return; setData((d) => ({ ...d, runLog: [...d.runLog, { id: uid(), date: form.date, type: form.type, dist, sec, pace: sec / dist }] })); setForm((f) => ({ ...f, dist: "", time: "", typeTouched: false, type: plannedRunFor(f.date, data.illnessLog, data.planEndMode)?.type ?? "int" })); }; // typeTouched wird nicht gespeichert; nach Add erneut aus Plan vorbelegen
  const { undo, trigger: triggerUndo, doUndo } = useUndoDelete(setData);
  const remove = (id) => {
    const item = data.runLog.find((r) => r.id === id);
    setData((d) => ({ ...d, runLog: d.runLog.filter((r) => r.id !== id) }));
    if (item) triggerUndo("Lauf", "runLog", item);
  };
  // V3: 5K-Prognose nur aus Qualitäts-Efforts — Easy/Lange bewusst ausgeschlossen,
  // bei Intervallen/Tempo zählen nur die Arbeitsrunden aus den CSV-Splits (Pausen rausgerechnet)
  const predData = useMemo(() => {
    const preds = data.runLog
      .map((r) => { const q = qualityEffort(r); const pred = q ? riegel5k(q.dist, q.sec) : null; return pred ? { iso: r.date, date: fmtISOshort(r.date), pred, basis: q.basis, dist: q.dist, pace: q.sec / q.dist } : null; })
      .filter(Boolean).sort((a, b) => a.iso.localeCompare(b.iso));
    let best = Infinity, bestInfo = null; const trend = [];
    preds.forEach((p) => { const isRecord = p.pred < best; if (isRecord) { best = p.pred; bestInfo = p; } trend.push({ date: p.date, pred: best, isRecord }); });
    const raw = preds.map((p) => ({ date: p.date, pred: p.pred }));
    return { trend, raw, bestInfo };
  }, [data.runLog]);
  const predTrend = predData.trend, predRaw = predData.raw, bestInfo = predData.bestInfo;
  const [predView, setPredView] = useState("best"); // "best" = Bestwert-Verlauf, "raw" = Prognose pro Lauf
  const bestPred = bestInfo ? bestInfo.pred : null;
  // Bestzeiten: schnellste effektive Pace je Lauftyp (Intervalle/Tempo = Arbeitsrunden-Pace)
  const pbs = useMemo(() => { const m = {}; data.runLog.forEach((r) => { const p = effectivePace(r); if (!m[r.type] || p < m[r.type].p) m[r.type] = { p, r }; }); return m; }, [data.runLog]);
  const pbIds = useMemo(() => new Set(Object.values(pbs).filter((e) => data.runLog.filter((x) => x.type === e.r.type).length >= 2).map((e) => e.r.id)), [pbs, data.runLog]);
  const confirmPending = () => {
    if (!pending) return;
    const key = `${pending.date}|${pending.dist.toFixed(1)}`;
    const exists = data.runLog.some((r) => `${r.date}|${r.dist.toFixed(1)}` === key);
    if (exists) { setImportMsg({ ok: false, text: "Für dieses Datum und diese Distanz existiert bereits ein Eintrag." }); return; }
    const { laps, typeTouched, ...rest } = pending; // typeTouched ist nur UI-Zustand, wird nicht gespeichert
    const entry = { id: uid(), imported: true, ...rest, ...(laps && laps.length > 1 ? { laps } : {}) };
    setData((d) => ({ ...d, runLog: [...d.runLog, entry] }));
    setImportMsg({ ok: true, text: `Lauf vom ${fmtISOshort(pending.date)} übernommen${laps?.length > 1 ? ` (inkl. ${laps.length} Runden-Splits)` : ""}.` });
    setPending(null);
  };
  const discardPending = () => { setPending(null); setImportMsg(null); };
  // v3.3: Herkunft des Typ-Vorschlags für die Vorschau-Karte (Plan-Treffer vs. Pace-Schätzung)
  const pendingAssign = pending ? assignRunType(pending.date, pending.pace, pending.dist, pending.sec, data.illnessLog, data.planEndMode) : null;

  const handleCsv = (file) => {
    if (!file) return;
    Papa.parse(file, {
      header: true, skipEmptyLines: true,
      complete: (res) => {
        try {
          const rows = res.data || [];
          if (!rows.length) { setImportMsg({ ok: false, text: "Die Datei enthält keine Daten." }); return; }
          const fields = (res.meta && res.meta.fields) || Object.keys(rows[0]);
          const hasDateCol = fields.some((f) => /datum|date/i.test(f));

          // ---- Fall A: Aktivitäten-Liste (mehrere Läufe, mit Datumsspalte) ----
          if (hasDateCol) {
            const parsed = [];
            rows.forEach((row) => {
              const actType = pickCol(row, ["aktivitätstyp", "activity type", "typ", "sport"]);
              if (actType && !/lauf|run|jog/i.test(String(actType))) return; // nur Läufe übernehmen
              const date = parseGarminDate(pickCol(row, ["datum", "date"]));
              const distRaw = pickCol(row, ["distanz", "distance", "strecke"]);
              const timeRaw = pickCol(row, ["zeit", "time", "dauer", "duration"]);
              if (!date || !distRaw) return;
              const dist = parseNumberLoose(distRaw);
              const sec = parseTimeToSec(timeRaw);
              if (!dist || isNaN(dist) || !sec) return;
              const pace = sec / dist;
              const a = assignRunType(date, pace, dist, sec, data.illnessLog, data.planEndMode); // v3.3: Plan gewinnt, Pace nur Fallback
              parsed.push({ date, dist: Math.round(dist * 100) / 100, sec, pace, type: a.type, _assign: a, ...extractExtras(row) });
            });
            if (parsed.length) {
              setData((d) => {
                const exists = new Set(d.runLog.map((r) => `${r.date}|${r.dist.toFixed(1)}`));
                const freshRaw = parsed.filter((p) => !exists.has(`${p.date}|${p.dist.toFixed(1)}`));
                // Statistik bewusst nur über tatsächlich importierte Läufe (Duplikate zählen nicht mit)
                const withExtras = freshRaw.filter((p) => p.hrAvg).length;
                const fromPlan = freshRaw.filter((p) => p._assign.source === "plan").length;
                const byPace = freshRaw.length - fromPlan;
                const mismatchDates = freshRaw.filter((p) => p._assign.mismatch).map((p) => fmtISOshort(p.date));
                const fresh = freshRaw.map(({ _assign, ...p }) => ({ id: uid(), imported: true, ...p }));
                const typeBits = !fresh.length ? ""
                  : fromPlan
                    ? ` Lauftyp bei ${fromPlan === fresh.length ? "allen" : fromPlan} aus dem Trainingsplan übernommen${byPace ? `, bei ${byPace} per Pace geschätzt (kein Plan-Lauf an dem Tag)` : ""}.`
                    : ` Lauftyp per Pace geschätzt — an diesen Tagen steht kein Lauf im Plan.`;
                setImportMsg({ ok: true, text: `${fresh.length} Läufe importiert (${parsed.length - fresh.length} Duplikate übersprungen).${typeBits}${withExtras ? ` Bei ${withExtras} wurden zusätzlich Herzfrequenz-/Höhendaten übernommen.` : ""}${mismatchDates.length ? ` ⚠️ Bitte prüfen (${mismatchDates.join(", ")}): Das gelaufene Tempo passt nicht recht zum geplanten Lauftyp — Typ ggf. im Eintrag anpassen.` : ""}` });
                return { ...d, runLog: [...d.runLog, ...fresh] };
              });
              return;
            }
            // Datumsspalte vorhanden, aber nichts brauchbares gefunden -> unten mit Fall B weiterprobieren
          }

          // ---- Fall B: Garmin-Einzelaktivitäts-/Rundenexport (Warm-up/Laufabschnitte/Cool-Down, ohne Datum) ----
          const summaryRow = findSummaryRow(rows) || rows[rows.length - 1];
          const distRaw = pickCol(summaryRow, ["distanz", "distance"]);
          const timeRaw = pickCol(summaryRow, ["gesamtzeit", "zeit", "time", "duration"]);
          const dist = parseNumberLoose(distRaw);
          const sec = parseTimeToSec(timeRaw);
          if (dist && sec && !isNaN(dist)) {
            const pace = sec / dist;
            const assigned = assignRunType(todayISO(), pace, dist, sec, data.illnessLog, data.planEndMode); // v3.3: Plan gewinnt, Pace nur Fallback
            const distR = Math.round(dist * 100) / 100;
            const extras = extractExtras(summaryRow);
            const laps = extractLaps(rows);
            setPending({ date: todayISO(), type: assigned.type, typeTouched: false, dist: distR, sec, pace, laps, ...extras });
            const bits = [];
            if (extras.hrAvg) bits.push(`Ø ${extras.hrAvg} bpm`);
            if (laps.length > 1) bits.push(`${laps.length} Runden-Splits`);
            setImportMsg({ ok: true, text: `Einzelaktivität erkannt (Rundenexport ohne Datum): ${distR} km in ${secToMMSS(sec)}${bits.length ? " · " + bits.join(" · ") : ""}. Vorschau unten prüfen — Datum setzen und übernehmen.` });
            return;
          }
          setImportMsg({ ok: false, text: "Keine Lauf-Daten gefunden. Unterstützt werden Garmin-Aktivitätenlisten (mit Datumsspalte) und Garmin-Einzelaktivitäts-/Rundenexporte (mit \u201eÜbersicht\u201c-Zeile)." });
        } catch (_) { setImportMsg({ ok: false, text: "Datei konnte nicht gelesen werden." }); }
      },
      error: () => setImportMsg({ ok: false, text: "Datei konnte nicht gelesen werden." }),
    });
  };

  const series = useMemo(() => data.runLog.filter((r) => r.type === type).sort((a, b) => a.date.localeCompare(b.date)).map((r) => ({ date: fmtISOshort(r.date), pace: Math.round(effectivePace(r)) })), [data.runLog, type]); // Intervalle/Tempo: Arbeitsrunden-Pace
  const target = targetPaceSec(type, effectiveWeek(currentWeek, data.illnessLog));
  const list = [...data.runLog].sort((a, b) => b.date.localeCompare(a.date)).slice(0, 8);
  const economySeries = useMemo(() => data.runLog.filter((r) => r.cadence).sort((a, b) => a.date.localeCompare(b.date)).map((r) => ({ date: fmtISOshort(r.date), cadence: r.cadence, stride: r.strideLen ?? null })), [data.runLog]);
  // V3: Pace-HF-Verhältnis & VO2max-Schätzung — nur ruhige Läufe (Easy/Lang/Tempo-Block); Intervalle verzerren (HF-Trägheit)
  const hrMax = hrMaxTanaka(Number(data.nutrition?.profile?.age) || 18);
  const steadyPoints = useMemo(() => data.runLog
    .filter((r) => r.hrAvg && STEADY_RUN(r) && r.dist >= 1.5)
    .sort((a, b) => a.date.localeCompare(b.date))
    .map((r) => { const e = r.type === "tempo" ? runEffort(r) : null; return { iso: r.date, date: fmtISOshort(r.date), pace: e ? e.pace : r.pace, hr: e && e.hr ? e.hr : r.hrAvg }; }), [data.runLog]);
  const efSeries = useMemo(() => steadyPoints.map((p) => ({ date: p.date, ef: efMetersPerBeat(p.pace, p.hr) })).filter((p) => p.ef), [steadyPoints]);
  const vo2Series = useMemo(() => {
    const pts = steadyPoints.map((p) => { const est = estimateVo2max(p.pace, p.hr, hrMax); return est ? { iso: p.iso, date: p.date, est } : null; }).filter(Boolean);
    return pts.map((p, i, a) => ({ ...p, trend: Math.round(median(a.slice(Math.max(0, i - 4), i + 1).map((x) => x.est)) * 10) / 10 }));
  }, [steadyPoints, hrMax]);
  const vo2Now = vo2Series.length ? vo2Series[vo2Series.length - 1].trend : null;
  const vo2Arrow = useMemo(() => {
    if (vo2Series.length < 2) return "";
    const last = vo2Series[vo2Series.length - 1];
    const old = [...vo2Series].reverse().find((p) => p.iso <= isoAddDays(last.iso, -21));
    if (!old) return "";
    return last.trend - old.trend >= 1 ? "↑" : old.trend - last.trend >= 1 ? "↓" : "→";
  }, [vo2Series]);
  return (
    <div className="space-y-4">
      {/* V3: 5K-Prognose — nur Qualitäts-Efforts */}
      <Card className="p-4" style={{ background: "linear-gradient(135deg,#312e81 0%,#4f46e5 100%)", borderColor: "transparent" }}>
        <Eyebrow color="#c7d2fe">5K-Prognose · Riegel-Formel</Eyebrow>
        {bestPred ? (
          <>
            <div className="mt-1 flex items-baseline gap-2.5">
              <span className="font-num text-3xl font-bold text-white leading-none">{secToMMSS(bestPred)}</span>
              <span className="text-indigo-200 text-xs font-num">≈ {secToMMSS(Math.round(bestPred / 5))}/km</span>
            </div>
            <div className="text-[11px] text-indigo-200 mt-1.5 font-num">Basis: {bestInfo.basis} · {bestInfo.dist} km @ {secToMMSS(Math.round(bestInfo.pace))}/km ({bestInfo.date})</div>
            {vo2Now && <div className="text-[11px] text-indigo-200 mt-0.5">Fitness-Trend: VO₂max ≈ {vo2Now} {vo2Arrow}</div>}
            {predTrend.length >= 2 && (
              <>
                <div className="flex gap-1 mt-2 mb-1">
                  <button onClick={() => setPredView("best")} className={`text-[10px] px-2 py-0.5 rounded-full font-medium transition ${predView === "best" ? "bg-white text-indigo-700" : "bg-white/15 text-indigo-100"}`}>Bestwert</button>
                  <button onClick={() => setPredView("raw")} className={`text-[10px] px-2 py-0.5 rounded-full font-medium transition ${predView === "raw" ? "bg-white text-indigo-700" : "bg-white/15 text-indigo-100"}`}>Pro Lauf</button>
                </div>
                <div style={{ width: "100%", height: 100 }}><ResponsiveContainer>
                  <LineChart data={predView === "best" ? predTrend : predRaw} margin={{ top: 4, right: 6, left: -4, bottom: 0 }}>
                    <XAxis dataKey="date" tick={{ fontSize: 9, fill: "#a5b4fc" }} axisLine={false} tickLine={false} />
                    <YAxis reversed domain={["dataMin - 20", "dataMax + 20"]} tickFormatter={secToMMSS} tick={{ fontSize: 9, fill: "#a5b4fc" }} width={46} axisLine={false} tickLine={false} />
                    <Tooltip formatter={(v) => [secToMMSS(v), predView === "best" ? "Bestwert" : "Prognose"]} {...tipProps(true)} />
                    <Line type={predView === "best" ? "stepAfter" : "linear"} dataKey="pred" stroke="#fff" strokeWidth={2.5} dot={predView === "best" ? (p) => (p.payload.isRecord ? <circle key={p.payload.date} cx={p.cx} cy={p.cy} r={2.5} fill="#fff" /> : null) : { r: 2.5, fill: "#fff" }} />
                  </LineChart>
                </ResponsiveContainer></div>
                <div className="text-[10px] text-indigo-200/70 mt-1">{predView === "best" ? "Zeigt deinen bisher schnellsten Prognosewert — bleibt flach, solange kein neuer Rekord dazukommt." : "Zeigt die Prognose jedes einzelnen Qualitäts-Laufs, auch wenn sie schlechter als dein Bestwert war."}</div>
              </>
            )}
            <div className="text-[10px] text-indigo-200/80 mt-1.5 leading-relaxed">Zählt nur echte Qualitäts-Efforts: 5K-Test, Tempo-Block, Intervall-Arbeitsrunden (Pausen rausgerechnet). Easy/Lange fließen bewusst nicht ein.</div>
          </>
        ) : (
          <div className="text-xs text-indigo-200 mt-1.5 leading-relaxed">Noch keine Qualitätseinheit erfasst. Die Prognose nutzt bewusst nur Intervalle/Tempo (Arbeitsrunden aus dem Garmin-Import) oder den 5K-Test — Easy-Läufe sagen nichts über deine Rennzeit aus und bleiben draußen.</div>
        )}
      </Card>

      {/* Garmin-Import */}
      <Card className="p-4" style={panel(dark, "#6366f1", "#eef2ff", "#c7d2fe")}>
        <div className="flex items-start gap-2">
          <Upload size={15} className="text-indigo-500 mt-0.5 shrink-0" />
          <p className="text-xs text-indigo-900 leading-relaxed"><b>Garmin-Import.</b> Exportiere in Garmin Connect deine Aktivitäten (oder eine Einzelaktivität mit Runden) als CSV und lade sie hier hoch. Neben Datum, Distanz und Zeit werden — sofern vorhanden — auch Herzfrequenz, Schrittfrequenz, Höhenmeter und Runden-Splits übernommen. Der Lauftyp wird automatisch aus deinem Trainingsplan übernommen (Datum → geplante Einheit); nur wenn dort kein Lauf steht, wird er anhand der Pace geschätzt.</p>
        </div>
        <input ref={fileRef} type="file" accept=".csv,text/csv" className="hidden" onChange={(e) => { handleCsv(e.target.files?.[0]); e.target.value = ""; }} />
        <button onClick={() => fileRef.current?.click()} className="mt-2.5 w-full rounded-xl border-2 border-dashed border-indigo-300 bg-white/60 py-2.5 text-sm font-semibold text-indigo-700 flex items-center justify-center gap-1.5"><Upload size={15} /> CSV-Datei auswählen</button>
        {importMsg && <div className={`mt-2 rounded-lg px-2.5 py-1.5 text-[11px] leading-relaxed ${importMsg.ok ? "bg-teal-100 text-teal-800" : "bg-rose-100 text-rose-700"}`}>{importMsg.text}</div>}
      </Card>

      {/* Vorschau eines erkannten Einzellaufs (Rundenexport ohne Datum) */}
      {pending && (
        <Card className="p-4" style={panel(dark, "#f59e0b", "#fffbeb", "#fcd34d")}>
          <Eyebrow color="#b45309">Vorschau · Datum bestätigen</Eyebrow>
          <div className="mt-2 grid grid-cols-2 gap-2.5">
            <input type="date" value={pending.date} onChange={(e) => setPending((p) => { const nd = e.target.value; const a = assignRunType(nd, p.pace, p.dist, p.sec, data.illnessLog, data.planEndMode); return { ...p, date: nd, ...(p.typeTouched ? {} : { type: a.type }) }; })} className="tc-input col-span-2" />
            <select value={pending.type} onChange={(e) => setPending({ ...pending, type: e.target.value, typeTouched: true })} className="tc-input col-span-2">{RUN_OPTS.map(([v, l]) => <option key={v} value={v}>{l}</option>)}</select>
          </div>
          {pendingAssign && !pending.typeTouched && (
            <div className="mt-2 text-[10.5px] leading-relaxed" style={{ color: safeText(dark, "#92400e") }}>
              {pendingAssign.source === "plan"
                ? `Typ aus dem Trainingsplan übernommen (Woche ${pendingAssign.planInfo.week}, ${DAY_FULL[pendingAssign.planInfo.day]}: ${RUN_TYPE[pendingAssign.planInfo.type].label}).`
                : "An diesem Tag steht kein Lauf im Plan — Typ per Pace geschätzt."}
              {pendingAssign.mismatch && ` ⚠️ Das gelaufene Tempo wirkt eher nach „${RUN_TYPE[pendingAssign.paceGuess].label}“ — bitte kurz prüfen.`}
            </div>
          )}
          <div className="mt-2.5 font-num text-sm text-slate-800">{pending.dist} km · {secToMMSS(pending.sec)} · {secToMMSS(pending.pace)}/km</div>
          <div className="mt-2 flex flex-wrap gap-1.5">
            {pending.hrAvg && <MiniBadge icon={<HeartPulse size={11} />} text={`Ø ${pending.hrAvg} bpm`} color="#e11d48" />}
            {pending.hrMax && <MiniBadge text={`Max ${pending.hrMax} bpm`} color="#e11d48" />}
            {pending.cadence && <MiniBadge text={`${pending.cadence} spm`} color="#0d9488" />}
            {pending.strideLen && <MiniBadge text={`${pending.strideLen} m Schritt`} color="#0d9488" />}
            {pending.elevGain != null && pending.elevGain > 0 && <MiniBadge icon={<Mountain size={11} />} text={`+${pending.elevGain} m`} color="#78350f" />}
            {pending.calories && <MiniBadge text={`${pending.calories} kcal`} color="#7c3aed" />}
            {pending.laps?.length > 1 && <MiniBadge text={`${pending.laps.length} Runden`} color="#4338ca" />}
          </div>
          <div className="mt-3 flex gap-2">
            <button onClick={discardPending} className="flex-1 rounded-xl bg-white border border-amber-300 text-amber-800 py-2 text-xs font-semibold">Verwerfen</button>
            <button onClick={confirmPending} className="flex-1 rounded-xl bg-amber-500 text-white py-2 text-xs font-semibold flex items-center justify-center gap-1.5"><Check size={14} /> Übernehmen</button>
          </div>
        </Card>
      )}

      <Card className="p-4"><Eyebrow color="#4338ca">Zielpace aktuell (Woche {currentWeek})</Eyebrow>
        <div className="mt-2 grid grid-cols-2 gap-2 text-sm">
          <div className="rounded-xl bg-indigo-50 p-2.5"><div className="text-[11px] text-indigo-500 font-semibold">Intervalle</div><div className="font-num font-bold text-indigo-900">{secToMMSS(targetPaceSec("int", effectiveWeek(currentWeek, data.illnessLog)))}/km</div></div>
          <div className="rounded-xl bg-indigo-50 p-2.5"><div className="text-[11px] text-indigo-500 font-semibold">Tempo</div><div className="font-num font-bold text-indigo-900">{secToMMSS(targetPaceSec("tempo", effectiveWeek(currentWeek, data.illnessLog)))}/km</div></div>
        </div>
      </Card>
      <Card className="p-4"><Eyebrow>Lauf manuell eintragen</Eyebrow>
        <div className="mt-3 grid grid-cols-2 gap-2.5">
          <input type="date" value={form.date} onChange={(e) => setForm((f) => { const nd = e.target.value; return { ...f, date: nd, ...(f.typeTouched ? {} : { type: plannedRunFor(nd, data.illnessLog, data.planEndMode)?.type ?? "int" }) }; })} className="tc-input col-span-2" />
          <select value={form.type} onChange={(e) => setForm({ ...form, type: e.target.value, typeTouched: true })} className="tc-input col-span-2">{RUN_OPTS.map(([v, l]) => <option key={v} value={v}>{l}</option>)}</select>
          {(() => { const fp = plannedRunFor(form.date, data.illnessLog, data.planEndMode); return (
            <div className="col-span-2 -mt-1 text-[10.5px] leading-relaxed" style={{ color: safeText(dark, fp ? "#4338ca" : "#94a3b8") }}>
              {fp && !form.typeTouched
                ? `Aus Plan vorbelegt: Woche ${fp.week}, ${DAY_FULL[fp.day]} — ${RUN_TYPE[fp.type].label}. Bei Bedarf oben anpassen.`
                : fp
                  ? `Plan an dem Tag: ${RUN_TYPE[fp.type].label} (du hast den Typ selbst gewählt).`
                  : "Kein Lauf im Plan an dem Tag — Typ frei wählbar."}
            </div>
          ); })()}
          <input inputMode="decimal" placeholder="Distanz (km)" value={form.dist} onChange={(e) => setForm({ ...form, dist: e.target.value })} className="tc-input" />
          <input inputMode="numeric" placeholder="Zeit (mm:ss)" value={form.time} onChange={(e) => setForm({ ...form, time: e.target.value })} className="tc-input font-num" />
        </div>
        <button onClick={add} className="mt-3 w-full rounded-xl bg-indigo-600 text-white py-2.5 text-sm font-semibold flex items-center justify-center gap-1.5"><Plus size={16} /> Hinzufügen</button>

function SyncCard({ data, setData, savedAt, onRefresh, refreshing }) {
  const [open, setOpen] = useState(false);
  const [text, setText] = useState("");
  const [msg, setMsg] = useState(null);
  const fileRef = useRef(null);

  const doExport = () => {
    const json = buildBackupPayload(data);
    const downloaded = downloadBackup(json);
    setText(json);
    let copied = false;
    try { if (navigator.clipboard?.writeText) { navigator.clipboard.writeText(json); copied = true; } } catch (_) {}
    setMsg({ ok: true, text: (downloaded ? "Datei heruntergeladen" : "Datei-Download nicht möglich") + (copied ? " und in die Zwischenablage kopiert." : ".") + " Auf dem anderen Gerät unten bei „Importieren“ einfügen oder die Datei öffnen." });
    setOpen(true);
  };
  const doImportText = () => {
    try {
      const parsed = JSON.parse(text);
      const incoming = parsed && parsed.data ? parsed.data : parsed;
      if (!incoming || typeof incoming !== "object") throw new Error("ungültig");
      setData({ ...DEFAULT_DATA, ...incoming });
      setMsg({ ok: true, text: "Daten übernommen — dieses Gerät ist jetzt auf dem importierten Stand." });
      setText("");
    } catch (_) { setMsg({ ok: false, text: "Konnte den Text nicht als Sicherung lesen. Bitte den vollständigen Export-Text/-Datei verwenden." }); }
  };
  const handleFile = (file) => {
    if (!file) return;
    const reader = new FileReader();
    reader.onload = () => setText(String(reader.result || ""));
    reader.readAsText(file);
  };

  return (
    <Card className="p-4">
      <div className="flex items-center justify-between gap-2">
        <Eyebrow>Geräteübergreifend nutzen</Eyebrow>
        {savedAt && <span className="text-[10px] text-slate-400 font-num shrink-0">gespeichert {savedAt.toLocaleTimeString("de-DE", { hour: "2-digit", minute: "2-digit" })}</span>}
      </div>
      <p className="text-xs text-slate-500 leading-relaxed mt-1.5">
        Deine Daten werden automatisch mit deinem Claude-Konto gespeichert — auf iPhone und iPad mit demselben Konto (und derselben Unterhaltung) siehst du in der Regel denselben Stand. Falls nicht, überträgst du hier per Sicherungsdatei/-text geräteunabhängig.
      </p>
      <div className="mt-3 flex gap-2">
        <button onClick={onRefresh} disabled={refreshing} className="flex-1 rounded-xl bg-slate-100 text-slate-700 py-2.5 text-xs font-semibold flex items-center justify-center gap-1.5 disabled:opacity-50"><RefreshCw size={13} className={refreshing ? "animate-spin" : ""} /> Aktualisieren</button>
        <button onClick={doExport} className="flex-1 rounded-xl bg-indigo-600 text-white py-2.5 text-xs font-semibold flex items-center justify-center gap-1.5"><Download size={13} /> Exportieren</button>
        <button onClick={() => setOpen((o) => !o)} className="flex-1 rounded-xl bg-slate-100 text-slate-700 py-2.5 text-xs font-semibold flex items-center justify-center gap-1.5"><Upload size={13} /> Importieren</button>
      </div>
      {msg && <div className={`mt-2 rounded-lg px-2.5 py-1.5 text-[11px] leading-relaxed ${msg.ok ? "bg-teal-100 text-teal-800" : "bg-rose-100 text-rose-700"}`}>{msg.text}</div>}
      {open && (
        <div className="mt-3 space-y-2">
          <input ref={fileRef} type="file" accept=".json,application/json" className="hidden" onChange={(e) => { handleFile(e.target.files?.[0]); e.target.value = ""; }} />
          <button onClick={() => fileRef.current?.click()} className="w-full rounded-xl border-2 border-dashed border-slate-300 py-2.5 text-xs font-semibold text-slate-600 flex items-center justify-center gap-1.5"><Upload size={13} /> Sicherungsdatei wählen</button>
          <textarea value={text} onChange={(e) => setText(e.target.value)} rows={4} placeholder="…oder Sicherungstext hier einfügen (vom anderen Gerät kopiert)" className="tc-input font-num" style={{ fontSize: 11 }} />
          <button onClick={doImportText} className="w-full rounded-xl bg-teal-600 text-white py-2.5 text-xs font-semibold flex items-center justify-center gap-1.5"><Copy size={13} /> Text übernehmen</button>
        </div>
      )}
    </Card>
  );
}

        </div>
      )}
    </Card>

function RecoveryTab({ data, setData, currentWeek }) {
  const dark = useDark();
  const [form, setForm] = useState({ date: todayISO(), sleep: "", readiness: 0, soreness: 0, note: "", napMin: "", napTime: "frueh_nachmittag", napFeeling: "neutral" });
  const save = () => {
    if (!form.sleep && !form.readiness && !form.napMin) return;
    const hasNap = Number(form.napMin) > 0;
    const entry = {
      id: uid(), date: form.date,
      sleep: Number(String(form.sleep).replace(",", ".")) || 0,
      readiness: form.readiness || 3,
      soreness: form.soreness || 0,
      note: form.note.trim(),
      ...(hasNap ? { napMin: Number(form.napMin), napTime: form.napTime, napFeeling: form.napFeeling } : {}),
    };
    setData((d) => ({ ...d, recoveryLog: [...d.recoveryLog.filter((r) => r.date !== form.date), entry] }));
    setForm({ date: todayISO(), sleep: "", readiness: 0, soreness: 0, note: "", napMin: "", napTime: "frueh_nachmittag", napFeeling: "neutral" });
  };
  const { undo, trigger: triggerUndo, doUndo } = useUndoDelete(setData);
  const remove = (id) => {
    const item = data.recoveryLog.find((r) => r.id === id);
    setData((d) => ({ ...d, recoveryLog: d.recoveryLog.filter((r) => r.id !== id) }));
    if (item) triggerUndo("Check-in", "recoveryLog", item);
  };
  const sorted = [...data.recoveryLog].sort((a, b) => a.date.localeCompare(b.date));
  const recents = [...sorted].reverse();
  const avgSleep = recents.length ? recents.slice(0, 5).reduce((s, r) => s + (r.sleep || 0), 0) / Math.min(5, recents.length) : null;
  const latest = recents[0];
  const weekLoad = plannedWeekLoad(currentWeek, data.vbSlots, data.illnessLog, data.planEndMode);
  const prios = data.prios?.length === 3 ? data.prios : DEFAULT_DATA.prios;
  const status = readinessStatus(weekLoad, avgSleep, latest?.readiness, SPORT[prios[2]].label, napFromEntry(latest));
  const sleepSeries = sorted.slice(-14).map((r) => ({ date: fmtISOshort(r.date), sleep: r.sleep }));
  const tWeek = planWeekOf(TOURNAMENT);
  const weekDays = useMemo(() => {
    const monday = mondayOfWeek(currentWeek);
    return buildWeekPlan(currentWeek, undefined, data.vbSlots, data.illnessLog, data.planEndMode).map((d, i) => {
      const row = { day: d.day, isToday: isSameDay(new Date(monday.getTime() + i * DAY_MS), new Date()), list: d.items.filter((it) => it.load > 0).map((it) => ({ sport: it.sport, label: it.label, load: it.load || 0, done: !!data.done[it.uid], excused: !data.done[it.uid] && it.illness === "sick" })) };
      ["vb", "run", "strength"].forEach((s) => { row[s + "Done"] = 0; row[s + "Excused"] = 0; row[s + "Open"] = 0; });
      row.list.forEach((it) => { row[it.sport + (it.done ? "Done" : it.excused ? "Excused" : "Open")] += it.load; });
      return row;
    });
  }, [currentWeek, data.vbSlots, data.done, data.illnessLog, data.planEndMode]);
  const totalW = planTotalWeeks(data.illnessLog, data.planEndMode);
  const meso = Array.from({ length: totalW }, (_, i) => {
    const w = i + 1;
    const bySport = weekLoadBySport(w, data.vbSlots, data.illnessLog, data.planEndMode);
    // Erledigt/Entschuldigt je Sportart — nur für vergangene/aktuelle Wochen; Zukunft = reiner Plan (alles "offen")
    const doneBy = { vb: 0, run: 0, strength: 0 }, excusedBy = { vb: 0, run: 0, strength: 0 };
    if (w <= currentWeek) buildWeekPlan(w, undefined, data.vbSlots, data.illnessLog, data.planEndMode).forEach((d) => d.items.forEach((it) => {
      if (doneBy[it.sport] == null) return;
      if (data.done[it.uid]) doneBy[it.sport] += it.load || 0;
      else if (it.illness === "sick") excusedBy[it.sport] += it.load || 0;
    }));
    const effW = w >= totalW ? TOTAL_WEEKS : effectiveWeek(w, data.illnessLog);
    const row = { w: `${w}`, current: w === currentWeek, future: w > currentWeek, kind: w >= totalW ? "Testwoche" : weekMeta(w).hasT ? "Turnierwoche" : REGEN[effW] ? "Regeneration" : "regulär" };
    ["vb", "run", "strength"].forEach((s) => {
      const doneV = Math.min(doneBy[s], bySport[s]);
      const excusedV = Math.min(excusedBy[s], Math.max(bySport[s] - doneV, 0));
      row[s + "Done"] = doneV; row[s + "Excused"] = excusedV; row[s + "Open"] = Math.max(bySport[s] - doneV - excusedV, 0); row[s + "Plan"] = bySport[s];
    });

}

        <div className="flex gap-1.5">{[["cut", "Abnehmen"], ["maintain", "Halten"], ["gain", "Aufbauen"]].map(([v, l]) => <Btn key={v} active={f.goal === v} onClick={() => set("goal", v)}>{l}</Btn>)}</div>
      </div>
      {f.goal !== "maintain" && (
        <div>
          <div className="text-[11px] font-semibold text-slate-500 mb-1.5">Tempo ({f.goal === "cut" ? "kg weniger" : "kg mehr"} pro Woche)</div>
          <div className="flex gap-1.5">{(f.goal === "cut" ? [0.25, 0.5, 0.75] : [0.25, 0.5]).map((v) => <Btn key={v} active={parseNum(f.rate) === v} onClick={() => set("rate", v)}>{String(v).replace(".", ",")} kg</Btn>)}</div>
          {f.goal === "cut" && parseNum(f.rate) >= 0.75 && <div className="text-[10.5px] text-amber-700 mt-1.5 leading-relaxed">0,75 kg/Woche ist mit 3 Sportarten sehr ambitioniert — 0,5 ist meist der bessere Deal zwischen Fortschritt und Trainingsqualität.</div>}
        </div>
      )}
      <div>
        <div className="text-[11px] font-semibold text-slate-500 mb-1.5">Trainings-kcal anrechnen (Garmin-Läufe & manuelle Aktivität)</div>
        <div className="flex gap-1.5">{[[0, "0 %", "konservativ"], [50, "50 %", "empfohlen"], [100, "100 %", "voll"]].map(([v, l, sub]) => <Btn key={v} active={parseNum(f.addBack) === v} onClick={() => set("addBack", v)} sub={sub}>{l}</Btn>)}</div>
      </div>
      <div>
        <div className="text-[11px] font-semibold text-slate-500 mb-1.5">Zielgewicht <span className="text-slate-400 font-normal">(optional, kg)</span></div>
        <input inputMode="decimal" value={f.goalWeight} onChange={(e) => set("goalWeight", e.target.value)} className="tc-input font-num" placeholder="z. B. 72" />
      </div>
      {plan ? (
        <div className="rounded-xl bg-slate-100 p-3 space-y-1">
          <div className="text-[10px] font-semibold text-slate-500 uppercase tracking-widest">Dein Plan</div>
          <div className="flex items-baseline gap-2"><span className="font-num text-2xl font-bold text-slate-900">{plan.target}</span><span className="text-[11px] text-slate-500">kcal/Tag {plan.delta ? `(${plan.delta > 0 ? "+" : ""}${plan.delta})` : "(Erhalt)"}</span></div>
          <div className="text-[10.5px] text-slate-500 font-num">Grundumsatz {plan.bmr} · Alltagsverbrauch {plan.tdee}{plan.calibrated ? " (kalibriert)" : ""} · an Trainingstagen automatisch mehr</div>
          {plan.belowFloor && <div className="text-[10.5px] text-amber-700 leading-relaxed">Ziel auf die Untergrenze {plan.floor} kcal angehoben — schneller abzunehmen würde Training & Gesundheit untergraben.</div>}
          {plan.calibrated && <button onClick={() => set("tdeeManual", null)} className="text-[10px] font-semibold text-slate-400 underline">Kalibrierung zurücksetzen (zurück zur Formel)</button>}
        </div>
      ) : <div className="rounded-xl bg-slate-100 p-3 text-[11px] text-slate-500">Alter, Größe und Gewicht eintragen — dann erscheint hier dein berechnetes Tagesziel.</div>}
      <button onClick={() => plan && onSave(f)} disabled={!plan} className="w-full rounded-xl bg-indigo-600 text-white py-2.5 text-sm font-semibold disabled:opacity-40">{submitLabel}</button>
    </div>
  );
}
function NutritionTab({ data, setData, plan }) {
  const dark = useDark();
  const nut = { ...NUT_DEFAULT, ...(data.nutrition || {}), profile: { ...NUT_DEFAULT.profile, ...((data.nutrition || {}).profile || {}) } };

  const saveProfile = (profile) => setData((d) => {
    const cur = { ...NUT_DEFAULT, ...(d.nutrition || {}) };
    const seed = (!cur.weightLog || !cur.weightLog.length) && parseNum(profile.weight) ? [{ id: uid(), date: todayISO(), kg: parseNum(profile.weight) }] : [];

  const sums = (d) => {
    const es = nut.intake.filter((e) => e.date === d);
    const inn = es.filter((e) => e.kind !== "out").reduce((s, e) => s + e.kcal, 0);
    const outM = es.filter((e) => e.kind === "out").reduce((s, e) => s + e.kcal, 0);
    const outG = garminKcalOn(data.runLog, d);
    return { inn, out: outM + outG, outG };
  };
  const targetOn = (d) => (kplan ? Math.round(kplan.target + (parseNum(nut.profile.addBack) / 100) * sums(d).out) : 0);

  if (!nut.setup || !kplan) return (
    <div className="space-y-4">
      <Card className="p-4">
        <Eyebrow color="#0d9488">Ernährung einrichten</Eyebrow>

  const t = todayISO();
  const ts = sums(t);
  const targetToday = targetOn(t);

  const barColor = pct > 1.07 ? "#d97706" : pct >= 0.9 ? "#0d9488" : "#6366f1";
  const todayKey = DAYS[(new Date().getDay() + 6) % 7];

    const k = Math.round(parseNum(entry.kcal)); if (!k) return;
    const e = { id: uid(), date: entry.date, kcal: k, kind: entry.kind, label: (entry.label || entry.tag || (entry.kind === "out" ? "Aktivität" : "Essen")).trim() };
    setData((d) => { const cur = d.nutrition; const presets = entry.savePreset ? [...cur.presets.filter((p) => !(p.label === e.label && p.kind === e.kind)), { id: uid(), label: e.label, kcal: k, kind: e.kind }].slice(-12) : cur.presets; return { ...d, nutrition: { ...cur, intake: [...cur.intake, e], presets } }; });
    setEntry((en) => ({ ...en, kcal: "", label: "", savePreset: false }));
  };
  const removeEntry = (id) => { const it = nut.intake.find((x) => x.id === id); setData((d) => ({ ...d, nutrition: { ...d.nutrition, intake: d.nutrition.intake.filter((x) => x.id !== id) } })); if (it) pushUndo("Eintrag", () => setData((d) => ({ ...d, nutrition: { ...d.nutrition, intake: [...d.nutrition.intake, it] } }))); };
  const removePreset = (id) => setData((d) => ({ ...d, nutrition: { ...d.nutrition, presets: d.nutrition.presets.filter((p) => p.id !== id) } }));
  const addWeight = () => { const kg = parseNum(wInput); if (!kg) return; setData((d) => ({ ...d, nutrition: { ...d.nutrition, weightLog: [...d.nutrition.weightLog.filter((w) => w.date !== t), { id: uid(), date: t, kg }], profile: { ...d.nutrition.profile, weight: String(kg) } } })); setWInput(""); };
  const removeWeight = (id) => { const it = nut.weightLog.find((x) => x.id === id); setData((d) => ({ ...d, nutrition: { ...d.nutrition, weightLog: d.nutrition.weightLog.filter((x) => x.id !== id) } })); if (it) pushUndo("Wiegung", () => setData((d) => ({ ...d, nutrition: { ...d.nutrition, weightLog: [...d.nutrition.weightLog, it] } }))); };

  const dayEntries = nut.intake.filter((e) => e.date === entry.date);
  const days14 = lastNDates(14).map((d) => { const s = sums(d); return { date: fmtISOshort(d), in: s.inn, target: targetOn(d) }; });
  let streakN = 0; { let d = t; if (!sums(d).inn) d = isoAddDays(d, -1); while (sums(d).inn > 0) { streakN++; d = isoAddDays(d, -1); } }
  const last7 = lastNDates(7);
  const hit7 = last7.filter((d) => { const s = sums(d); if (!s.inn) return false; const tg = targetOn(d); return Math.abs(s.inn - tg) <= 0.1 * tg; }).length;
  const logged7 = last7.map((d) => sums(d).inn).filter((v) => v > 0);
  const avg7 = logged7.length ? Math.round(logged7.reduce((a, b) => a + b, 0) / logged7.length) : null;
  const now = new Date(); const mo = new Date(now); mo.setDate(now.getDate() - ((now.getDay() + 6) % 7));
  const weekIso = Array.from({ length: 7 }, (_, i) => { const d = new Date(mo); d.setDate(mo.getDate() + i); return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, "0")}-${String(d.getDate()).padStart(2, "0")}`; });
  const weekBudget = weekIso.reduce((s, d) => s + (d <= t ? targetOn(d) : kplan.target), 0) - weekIso.reduce((s, d) => s + sums(d).inn, 0);

  let wTrend = null;

  // Realitäts-Check: echten Tagesverbrauch aus Essen + Gewichtsverlauf schätzen (21-Tage-Fenster)

          ))}

"" : tg })} className={`rounded-full px-2.5 py-1 text-[10.5px] font-semibold border ${entry.tag === tg ? "bg-teal-600 text-white border-transparent" : "bg-white text-slate-600 border-slate-200"}`}>{tg}</button>)}</div>}

Noch nötig: <b>{need.days} Log-Tage</b> und <b>{need.weighs} Wiegungen</b> (im 21-Tage-Fenster, erste und letzte Wiegung mind. 14 Tage auseinander).</div>
        )}
        <div className="mt-2"><InfoToggle label="Wie wird gerechnet?">Ø-Kalorienaufnahme minus Gewichtsänderung × 7700 kcal/kg über den Zeitraum. Nur aussagekräftig, wenn (fast) jeder Tag geloggt ist — Lücken verfälschen das Ergebnis nach unten.</InfoToggle></div>
      </Card>

const FIT_EPOCH_OFFSET = 631065600; // Unix-Sekunden am 1989-12-31T00:00Z (FIT-Zeitbasis)
function fitCrc16(bytes) {
  const T = [0x0000, 0xCC01, 0xD801, 0x1400, 0xF001, 0x3C00, 0x2800, 0xE401, 0xA001, 0x6C00, 0x7800, 0xB401, 0x5000, 0x9C01, 0x8801, 0x4400];
  let c = 0;
  for (let i = 0; i < bytes.length; i++) {
    const b = bytes[i];
    let t = T[c & 15]; c = (c >> 4) & 0x0FFF; c = c ^ t ^ T[b & 15];
    t = T[c & 15]; c = (c >> 4) & 0x0FFF; c = c ^ t ^ T[(b >> 4) & 15];
  }
  return c;
}
function fitBuf() {
  const a = [];
  const b = {
    a,
    u8(v) { a.push(v & 255); },
    u16(v) { b.u8(v); b.u8(v >> 8); },
    u32(v) { b.u16(v & 0xFFFF); b.u16((v >>> 16) & 0xFFFF); },
    str(s, len) { for (let i = 0; i < len; i++) b.u8(i < s.length ? s.charCodeAt(i) & 127 : 0); },
  };
  return b;
}
// Basistypen: enum 0x00 · uint16 0x84 · uint32 0x86 · string 0x07 · uint32z 0x8C
function fitDef(b, local, global, fields) { b.u8(0x40 | local); b.u8(0); b.u8(0); b.u16(global); b.u8(fields.length); fields.forEach((f) => { b.u8(f[0]); b.u8(f[1]); b.u8(f[2]); }); }
const WKT = { DUR_TIME: 0, DUR_DIST: 1, DUR_OPEN: 5, DUR_REPEAT: 6, TGT_SPEED: 0, TGT_HR: 1, TGT_OPEN: 2, INT_ACTIVE: 0, INT_REST: 1, INT_WARM: 2, INT_COOL: 3 };
function encodeFitWorkout(name, steps) {
  const b = fitBuf();
  fitDef(b, 0, 0, [[0, 1, 0x00], [1, 2, 0x84], [2, 2, 0x84], [3, 4, 0x8C], [4, 4, 0x86]]); // file_id: Typ 5 = Workout
  b.u8(0); b.u8(5); b.u16(255); b.u16(0); b.u32(1 + Math.floor(Math.random() * 0xFFFFFF)); b.u32(Math.floor(Date.now() / 1000) - FIT_EPOCH_OFFSET);
  fitDef(b, 1, 26, [[8, 16, 0x07], [4, 1, 0x00], [6, 2, 0x84]]); // workout: Name, Sport (1 = Laufen), Schrittzahl
  b.u8(1); b.str(name.slice(0, 15), 16); b.u8(1); b.u16(steps.length);
  fitDef(b, 2, 27, [[254, 2, 0x84], [0, 16, 0x07], [1, 1, 0x00], [2, 4, 0x86], [3, 1, 0x00], [4, 4, 0x86], [5, 4, 0x86], [6, 4, 0x86], [7, 1, 0x00]]); // workout_step
  steps.forEach((s, i) => { b.u8(2); b.u16(i); b.str((s.name || "").slice(0, 15), 16); b.u8(s.dur); b.u32((s.durVal || 0) >>> 0); b.u8(s.tgt); b.u32((s.tgtVal || 0) >>> 0); b.u32((s.low || 0) >>> 0); b.u32((s.high || 0) >>> 0); b.u8(s.int); });
  const data = b.a;
  const h = fitBuf();
  h.u8(14); h.u8(0x10); h.u16(2140); h.u32(data.length); ".FIT".split("").forEach((c) => h.u8(c.charCodeAt(0)));
  h.u16(fitCrc16(h.a)); // Header-CRC über die ersten 12 Bytes
  const all = [...h.a, ...data];
  const crc = fitCrc16(all);
  return new Uint8Array([...all, crc & 255, crc >> 8]);
}
const hrTgt = (lo, hi) => ({ tgt: WKT.TGT_HR, tgtVal: 0, low: lo + 100, high: hi + 100 }); // FIT-Konvention: bpm = Wert − 100
const paceTgt = (paceSec, pm = 0.05) => { const v = 1000000 / paceSec; return { tgt: WKT.TGT_SPEED, tgtVal: 0, low: Math.round(v * (1 - pm)), high: Math.round(v * (1 + pm)) }; }; // Geschwindigkeit in mm/s
const OPEN_TGT = { tgt: WKT.TGT_OPEN, tgtVal: 0, low: 0, high: 0 };
const parseHrRange = (hr) => { const m = /(\d+)\s*[–-]\s*(\d+)/.exec(hr || ""); return m ? [+m[1], +m[2]] : null; };
const parseMinutesAvg = (s) => { const m = /(\d+)(?:\s*[–-]\s*(\d+))?\s*Min/i.exec(s || ""); return m ? Math.round((+m[1] + (+(m[2] || m[1]))) / 2) : null; };
const parseReps = (s) => { const m = /(\d+)\s*[×x]\s*(\d+(?:[.,]\d+)?)\s*(m|km)\b/i.exec(s || ""); if (!m) return null; const d = parseFloat(m[2].replace(",", ".")); return { n: +m[1], distM: Math.round(m[3].toLowerCase() === "km" ? d * 1000 : d) }; };
const parsePauseSec = (s) => { const m = /(\d+(?:[.,]\d+)?)\s*Min\s*Pause/i.exec(s || ""); if (m) return Math.round(parseFloat(m[1].replace(",", ".")) * 60); const m2 = /(\d+)\s*s(?:ek)?\w*\s*Pause/i.exec(s || ""); return m2 ? +m2[1] : 120; };
// Plan-Session → strukturiertes Garmin-Workout (HF-/Pace-Zielbereiche & Wiederholungen voreingestellt)
function buildGarminWorkout(item, week) {
  const hr = parseHrRange(item.hr);
  const warm = { name: "Aufwärmen", int: WKT.INT_WARM, dur: WKT.DUR_TIME, durVal: 10 * 60 * 1000, ...OPEN_TGT };
  const cool = { name: "Auslaufen", int: WKT.INT_COOL, dur: WKT.DUR_TIME, durVal: 5 * 60 * 1000, ...OPEN_TGT };
  const short = { easy: "Easy", tempo: "Tempo", int: "Intervalle", long: "Lang", test: "5K-Test" }[item.type];
  if (!short) return null;
  const name = `W${week} ${short}`;
  let steps = [];
  if (item.type === "easy" || item.type === "long") {
    const min = parseMinutesAvg(item.dur) || 30;
    steps = [{ name: short, int: WKT.INT_ACTIVE, dur: WKT.DUR_TIME, durVal: min * 60 * 1000, ...(hr ? hrTgt(hr[0], hr[1]) : OPEN_TGT) }];
  } else if (item.type === "tempo") {
    const min = parseMinutesAvg(item.dur) || 20;
    steps = [warm, { name: "Tempo", int: WKT.INT_ACTIVE, dur: WKT.DUR_TIME, durVal: min * 60 * 1000, ...(hr ? hrTgt(hr[0], hr[1]) : paceTgt(targetPaceSec("tempo", week))) }, cool];
  } else if (item.type === "int") {
    const reps = parseReps(item.dur) || parseReps(item.desc) || { n: 5, distM: 800 };
    const pause = parsePauseSec(item.desc || item.dur);
    steps = [warm,
      { name: `${reps.distM} m`, int: WKT.INT_ACTIVE, dur: WKT.DUR_DIST, durVal: reps.distM * 100, ...paceTgt(targetPaceSec("int", week)) },
      { name: "Trabpause", int: WKT.INT_REST, dur: WKT.DUR_TIME, durVal: pause * 1000, ...OPEN_TGT },
      { name: "Wiederholen", int: WKT.INT_ACTIVE, dur: WKT.DUR_REPEAT, durVal: 1, tgt: WKT.TGT_OPEN, tgtVal: reps.n, low: 0, high: 0 },
      cool];
  } else if (item.type === "test") {
    steps = [warm, { name: "5 km Bestzeit", int: WKT.INT_ACTIVE, dur: WKT.DUR_DIST, durVal: 500000, ...OPEN_TGT }, cool];
  }
  return { name, steps };
}
function workoutRecipeLines(spec) {
  return spec.steps.map((s, idx) => {
    const n = idx + 1;
    if (s.dur === WKT.DUR_REPEAT) return `${n}. Wiederholen: Schritte ${s.durVal + 1}–${idx} ×${s.tgtVal}`;
    const d = (s.durVal || 0) / 100;
    const dur = s.dur === WKT.DUR_TIME ? `${Math.round(s.durVal / 60000)} Min` : s.dur === WKT.DUR_DIST ? (d >= 1000 ? `${d / 1000} km` : `${d} m`) : "offen";
    const tgt = s.tgt === WKT.TGT_HR ? `HF ${s.low - 100}–${s.high - 100}` : s.tgt === WKT.TGT_SPEED ? `Pace ${secToMMSS(Math.round(1000000 / s.high))}–${secToMMSS(Math.round(1000000 / s.low))}/km` : "locker";
    return `${n}. ${s.name} — ${dur}, ${tgt}`;
  });
}
function GarminExport({ item, week }) {
  const spec = useMemo(() => buildGarminWorkout(item, week), [item, week]);
  const [copied, setCopied] = useState(false);
  if (!spec) return null;
  const lines = workoutRecipeLines(spec);
  const download = () => { try { const bytes = encodeFitWorkout(spec.name, spec.steps); const blob = new Blob([bytes], { type: "application/octet-stream" }); const url = URL.createObjectURL(blob); const a = document.createElement("a"); a.href = url; a.download = spec.name.replace(/\s+/g, "_") + ".fit"; document.body.appendChild(a); a.click(); a.remove(); setTimeout(() => URL.revokeObjectURL(url), 4000); } catch (_) {} };
  const copy = async () => { try { await navigator.clipboard.writeText(spec.name + "\n" + lines.join("\n")); setCopied(true); setTimeout(() => setCopied(false), 2000); } catch (_) {} };
  return (
    <div className="rounded-xl bg-slate-100 p-2.5 space-y-2">
      <div className="flex items-center gap-1.5 text-[10px] font-semibold text-slate-500"><Watch size={12} /> Auf die Garmin-Uhr <span className="rounded bg-amber-100 text-amber-700 px-1.5 py-0.5">experimentell</span></div>
      <ul className="space-y-0.5">{lines.map((l, i) => <li key={i} className="text-[10.5px] text-slate-600 font-num">{l}</li>)}</ul>
      <div className="flex gap-1.5">
        <button onClick={download} className="flex-1 rounded-lg bg-slate-900 text-white py-2 text-[11px] font-semibold inline-flex items-center justify-center gap-1"><Download size={12} /> .FIT laden</button>
        <button onClick={copy} className="flex-1 rounded-lg bg-white border border-slate-200 text-slate-700 py-2 text-[11px] font-semibold inline-flex items-center justify-center gap-1"><Copy size={12} /> {copied ? "Kopiert ✓" : "Rezept kopieren"}</button>
      </div>
      <InfoToggle label="So kommt's auf die Forerunner 55">
        <b>Mit Kabel:</b> Uhr per USB an den Computer → Laufwerk „GARMIN“ → die .FIT-Datei in den Ordner <b>GARMIN/NEWFILES</b> kopieren → auswerfen. Auf der Uhr: Lauf-Profil → UP lang gedrückt → Training → Trainings → „{spec.name}“ wählen. HF-/Pace-Zielbereiche, Distanzen und Wiederholungen sind voreingestellt — die Uhr führt dich durch jede Stufe.<br />
        <b>Ohne Kabel:</b> „Rezept kopieren“ und in der Garmin-Connect-App unter Mehr → Training → Trainings 1:1 anlegen — synchronisiert automatisch auf die Uhr.<br />
        Der Datei-Export ist experimentell: Falls die Uhr das Workout nicht anzeigt, bitte den Connect-Weg nehmen.
      </InfoToggle>
    </div>

function SchoolTab({ data, setData }) {
  const dark = useDark();
  const [form, setForm] = useState({ titel: "", fach: "", faelligkeit: todayISO(), art: "hausaufgabe" });
  const [editingId, setEditingId] = useState(null);
  const resetForm = () => setForm((f) => ({ titel: "", fach: "", faelligkeit: todayISO(), art: f.art }));
  const save = () => {
    if (!form.titel.trim()) return;
    if (editingId) {
      setData((d) => ({ ...d, schoolLog: d.schoolLog.map((s) => (s.id === editingId ? { ...s, titel: form.titel.trim(), fach: form.fach.trim(), faelligkeit: form.faelligkeit, art: form.art } : s)) }));
      setEditingId(null);
    } else {
      const entry = { id: uid(), titel: form.titel.trim(), fach: form.fach.trim(), faelligkeit: form.faelligkeit, art: form.art, erledigt: false };
      setData((d) => ({ ...d, schoolLog: [...d.schoolLog, entry] }));
    }
    resetForm();
  };
  const startEdit = (item) => {
    setEditingId(item.id);
    setForm({ titel: item.titel, fach: item.fach, faelligkeit: item.faelligkeit, art: item.art });
  };
  const cancelEdit = () => {
    setEditingId(null);
    resetForm();
  };
  const toggleErledigt = (id) => setData((d) => ({ ...d, schoolLog: d.schoolLog.map((s) => (s.id === id ? { ...s, erledigt: !s.erledigt } : s)) }));
  const { undo, trigger: triggerUndo, doUndo } = useUndoDelete(setData);
  const remove = (id) => {
    const item = data.schoolLog.find((s) => s.id === id);
    setData((d) => ({ ...d, schoolLog: d.schoolLog.filter((s) => s.id !== id) }));
    if (item) triggerUndo(item.art === "klausur" ? "Klausur" : "Hausaufgabe", "schoolLog", item);
    if (editingId === id) cancelEdit();
  };
  const sorted = [...data.schoolLog].sort((a, b) => (a.erledigt === b.erledigt ? a.faelligkeit.localeCompare(b.faelligkeit) : a.erledigt ? 1 : -1));
  const openCount = data.schoolLog.filter((s) => !s.erledigt).length;
  const overdueCount = data.schoolLog.filter((s) => !s.erledigt && daysUntil(s.faelligkeit) < 0).length;

  return (
    <div className="space-y-4">
      <Card className="p-4">
        <Eyebrow>{editingId ? "Eintrag bearbeiten" : "Neu eintragen"}</Eyebrow>
        <div className="mt-3 space-y-2.5">
          <input placeholder="Was? (z. B. Mathe S. 42, Aufgabe 3)" value={form.titel} onChange={(e) => setForm({ ...form, titel: e.target.value })} className="tc-input" />
          <div className="grid grid-cols-2 gap-2.5">
            <div>
              <div className="text-[11px] font-semibold text-slate-500 mb-1.5">Fach (optional)</div>
              <input placeholder="z. B. Mathe" value={form.fach} onChange={(e) => setForm({ ...form, fach: e.target.value })} className="tc-input" />
            </div>
            <div>
              <div className="text-[11px] font-semibold text-slate-500 mb-1.5">Fällig bis</div>
              <input type="date" value={form.faelligkeit} onChange={(e) => setForm({ ...form, faelligkeit: e.target.value })} className="tc-input font-num" />
            </div>
          </div>
          <div className="flex gap-1.5">
            {SCHOOL_ART.map(([v, l]) => (
              <button key={v} onClick={() => setForm({ ...form, art: v })} className={`flex-1 rounded-lg px-2 py-2 text-[11px] font-semibold border transition ${form.art === v ? "bg-indigo-600 text-white border-transparent" : "bg-white text-slate-600 border-slate-200"}`}>{l}</button>
            ))}
          </div>
          <div className="flex gap-2">
            <button onClick={save} className="flex-1 rounded-xl bg-indigo-600 text-white py-2.5 text-sm font-semibold flex items-center justify-center gap-1.5">
              {editingId ? <Check size={16} /> : <Plus size={16} />} {editingId ? "Speichern" : "Eintragen"}
            </button>
            {editingId && (
              <button onClick={cancelEdit} className="rounded-xl border border-slate-200 text-slate-500 px-4 py-2.5 text-sm font-semibold">Abbrechen</button>
            )}
          </div>
        </div>
      </Card>

      <Card className="p-4">
        <div className="flex items-center justify-between mb-2">
          <Eyebrow>Einträge</Eyebrow>
          <span className="text-[11px] font-num text-slate-400">{openCount} offen{overdueCount ? ` · ${overdueCount} überfällig` : ""}</span>

                    {item.erledigt && <Check size={14} color="#fff" strokeWidth={3} />}

                    <div className={`text-sm font-semibold ${item.erledigt ? "line-through text-slate-400" : "text-slate-800"}`}>{item.titel}</div>

              );
            })}
          </div>
        ) : <div className="py-8 text-center text-xs text-slate-400">Noch nichts eingetragen.</div>}
      </Card>

      <UndoToast undo={undo} onUndo={doUndo} />
    </div>
  );
}
